unsigned hamdist(unsigned x, unsigned y)
{
  unsigned dist = 0, val = x ^ y; // XOR
 
  // Count the number of set bits
  while(val)
  {
    ++dist; 
    val &= val - 1;
  }
 
  return dist;
}
